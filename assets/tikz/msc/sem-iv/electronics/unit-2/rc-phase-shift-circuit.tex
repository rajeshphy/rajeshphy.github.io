\documentclass[border=8pt]{standalone}
\usepackage[american]{circuitikz}
\begin{document}
\begin{circuitikz}[font=\sffamily]
\node[op amp] (A) at (8.2,2.8) {};
\draw (A.+) -- ++(-.55,0) node[ground]{};
\draw (A.-) -- ++(-.60,0) coordinate(sum);
\fill (sum) circle (1.4pt);
\draw (A.out) -- ++(1.0,0) coordinate(vo) node[right]{$V_o$};

% The feedback resistor is the only connection between its named endpoints.
\draw (vo) -- ++(0,2.0) coordinate(fbOut);
\coordinate (fbSum) at (sum |- fbOut);
\draw (fbOut) to[R,l=$R_f$] (fbSum);
\draw (fbSum) -- (sum);

% Three loaded CR sections; R_g=R is the third shunt to virtual ground.
\draw (vo) -- ++(0,-2.8) coordinate(chainOut);
\draw (chainOut) to[C,l_=$C$] ++(-1.55,0) coordinate(n1)
 to[C,l_=$C$] ++(-1.55,0) coordinate(n2)
 to[C,l_=$C$] ++(-1.55,0) coordinate(vf);
\fill (n1) circle (1.2pt);
\fill (n2) circle (1.2pt);
\fill (vf) circle (1.4pt);
\draw (n1) to[R,l=$R$] ++(0,-1.8) node[ground]{};
\draw (n2) to[R,l=$R$] ++(0,-1.8) node[ground]{};
\draw (vf) -- ++(0,.40) coordinate(rgIn);
\coordinate (rgOut) at (rgIn |- sum);
\draw (rgIn) to[R,l={$R_g=R$}] (rgOut);
\draw (rgOut) -- (sum);
\node[left] at (vf) {$V_f$};
\node[above] at (5.4,5.55) {Three-section RC phase-shift oscillator};
\end{circuitikz}
\end{document}
