\begin{circuitikz}[font=\sffamily]
 \node[font=\bfseries] at (-4.6,2.55) {(a) Differential input};
 \draw (-4.6,0) node[op amp](A){};
 \draw (A.+) -- (-6.2,-.5) node[left] {$v_1$};
 \draw (A.-) -- (-6.2,.5) node[left] {$v_2$};
 \draw (A.out) -- (-2.95,0) node[right] {$v_o$};
 \node at (-4.6,-1.2) {$v_o=A_{OL}(v_1-v_2)$};

 \node[font=\bfseries] at (0,2.55) {(b) Inverting input};
 \draw (0,0) node[op amp](B){};
 \draw (B.+) -- (-1.3,-.5) node[ground]{};
 \draw (B.-) -- (-1.6,.5) node[left] {$v_i$};
 \draw (B.out) -- (1.55,0) node[right] {$v_o$};
 \node at (0,-1.2) {$v_o=-A_{OL}v_i$};

 \node[font=\bfseries] at (4.8,2.55) {(c) Non-inverting input};
 \draw (4.8,0) node[op amp](C){};
 \draw (C.-) -- (3.5,.5) node[ground]{};
 \draw (C.+) -- (3.2,-.5) node[left] {$v_i$};
 \draw (C.out) -- (6.35,0) node[right] {$v_o$};
 \node at (4.8,-1.2) {$v_o=A_{OL}v_i$};
 \draw[dashed] (-2.4,-1.4) -- (-2.4,2.9);
 \draw[dashed] (2.5,-1.4) -- (2.5,2.9);
 \node[draw,rounded corners,fill=red!4,align=center] at (0,-2.65)
 {With no feedback, even a small differential input normally drives\\
 the output to $+V_{sat}$ or $-V_{sat}$; these are comparator configurations.};
